PROCEDURE prc_overtime(p_emp_id IN NUMBER) IS
  v_hours NUMBER;
  v_rate  NUMBER;
BEGIN
  SELECT base_rate INTO v_rate
    FROM emp_pay WHERE emp_id = p_emp_id;

  IF v_hours > 40 THEN
    v_rate := v_rate * 1.5;
    IF v_hours > 60 THEN
      v_rate := v_rate * 2;
    END IF;
  END IF;

END;

DDL Conversion Preview

Oracle NUMBER → NUMERIC · VARCHAR2 → VARCHAR · SEQUENCE+TRIGGER → BIGSERIAL
ORACLE 19c · SOURCE
CREATE TABLE inventory_items (
  item_id           NUMBER(15) NOT NULL,
  item_code         VARCHAR2(50),
  description       VARCHAR2(500),
  unit_cost         NUMBER(18,4),
  quantity_on_hand  NUMBER(15,3),
  last_updated      DATE,
  created_by        VARCHAR2(30),
  CONSTRAINT pk_inv PRIMARY KEY (item_id)
);

-- Oracle sequence + trigger for auto-increment
CREATE SEQUENCE inv_seq
  START WITH 1 INCREMENT BY 1
  NOCACHE NOCYCLE;

CREATE OR REPLACE TRIGGER trg_inv_id
BEFORE INSERT ON inventory_items
FOR EACH ROW BEGIN
  SELECT inv_seq.NEXTVAL INTO :new.item_id
  FROM dual;
END;
POSTGRESQL 15 · TARGET Auto-converted
CREATE TABLE inventory_items (
  item_id          BIGSERIAL PRIMARY KEY,
  item_code        VARCHAR(50),
  description      TEXT,
  unit_cost        NUMERIC(18,4),
  quantity_on_hand NUMERIC(15,3),
  last_updated     TIMESTAMP,
  created_by       VARCHAR(30)
);

-- Conversion notes:
--   NUMBER(18,4)  → NUMERIC(18,4)
