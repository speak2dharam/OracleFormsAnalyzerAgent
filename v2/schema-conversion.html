CREATE TABLE inventory_items ( item_id NUMBER(15) NOT NULL, item_code VARCHAR2(50), description VARCHAR2(500), unit_cost NUMBER(18,4), quantity_on_hand NUMBER(15,3), last_updated DATE, created_by VARCHAR2(30), CONSTRAINT pk_inv PRIMARY KEY (item_id) ); -- Oracle sequence for auto-increment CREATE SEQUENCE inv_seq START WITH 1 INCREMENT BY 1 NOCACHE NOCYCLE; CREATE OR REPLACE TRIGGER trg_inv_id BEFORE INSERT ON inventory_items FOR EACH ROW BEGIN SELECT inv_seq.NEXTVAL INTO :new.item_id FROM dual; END;
